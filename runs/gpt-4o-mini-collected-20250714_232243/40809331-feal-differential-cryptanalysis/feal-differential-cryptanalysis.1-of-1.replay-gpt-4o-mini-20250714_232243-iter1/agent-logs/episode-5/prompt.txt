    # Initial key whitening
    left ^= key[0]
    right ^= key[1]

    # Pre-round mixing
    right ^= left

    # 4 rounds of Feistel network
    for round_num in range(4):
        mixed = f_function(right ^ key[round_num+2])
        temp = right
        right = left ^ mixed
        left = temp


    # Post-round mixing
    left ^= right

    # Swap and combine
    return merge(right, left)
